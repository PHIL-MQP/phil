def yawdiff(y1, y2):
    diff = y2 - y1
    if diff > np.pi:
        return diff - np.pi * 2;
    if diff < -np.pi:
        return diff + np.pi * 2;
    return diff;

    state[2] += drz
    states[mocap_idx] = state

vys = []
x_bias = 0 #.0385
y_bias = 0 #.041
x_scale = 1000
y_scale = 1000
for data in imu_data:
    ax = (data[0] - x_bias) * x_scale
    ay = (data[1] - y_bias) * y_scale
    dt_s = data[-1] - last_t
    vx += ax * dt_s
    vy += ay * dt_s
    x += vx * dt_s + 0.5 * ax * dt_s ** 2
    y += vy * dt_s + 0.5 * ay * dt_s ** 2
    last_t = data[-1]
    vxs.append(vx)
    vys.append(vy)
    xs.append(x)
    ys.append(y)
