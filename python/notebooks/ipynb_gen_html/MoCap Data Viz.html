def yawdiff(y1, y2):
    diff = y2 - y1
    if diff > np.pi:
        return diff - np.pi * 2;
    if diff < -np.pi:
        return diff + np.pi * 2;
    return diff;

def DoubleIntegrateAccelerometer(accelerometer_data, K, T, b):
    x = 0
    y = 0
    vx = 0
    vy = 0
    dt_s = .02
    xs = []
    ys = []
    vxs = []
    vys = []
    axs = []
    ays = []
    
    for data in accelerometer_data:
        a_s = data[3:6]
        a_o = T@K@(a_s+b).T
        ax = a_o[0][0]
        ay = a_o[1][0]
        
        vx += ax * dt_s
        vy += ay * dt_s
        x += vx * dt_s + 0.5 * ax * dt_s ** 2
        y += vy * dt_s + 0.5 * ay * dt_s ** 2
        axs.append(ax)
        ays.append(ay)
        vxs.append(vx)
        vys.append(vy)
        xs.append(x)
        ys.append(y)
    
    return xs, ys, vxs, vys, axs, ays
